var n1 = Number(window.prompt('Digite um número: ')) // Se lê assim: n1 vai receber(=) o prompt que abre uma janela, convertido para um numero.
var n2 = Number(window.prompt('Digite outro número: '))
var soma = n1 + n2
// window.alert('A soma dos valores é  ' + soma)
window.alert(`A soma entre ${n1} e ${n2} é igual a ${soma}`) // modo template String
// String(n) ou n.String() para converter numero em String.
// Number ou Number.parseFloat ou Number.parseInt usa-se para converter String em numero.
// Number deixa o JS descidir se é float ou int
